2. Client tracking:

3. Handler function:
   ```python
   async def handler(websocket):
       client_id = f"client-{len(clients)}"
       clients[websocket] = {'id': client_id, 'type': 'unknown'}
       print(f"[{client_id}] Connected (total: {len(clients)})")
       
       try:
           async for message in websocket:
               try:
                   data = json.loads(message)
                   cmd = data.get('cmd', 'unknown')
                   
                   # Identify client type
                   if cmd in ('init', 'score', 'settle', 'reset'):
                       clients[websocket]['type'] = 'controller'
                   
                   print(f"[{client_id}] Received: {cmd}")
                   
                   # Broadcast to all other clients
                   for client, info in list(clients.items()):
                       if client != websocket:
                           try:
                               await client.send(message)
                               print(f"  -> Forwarded to [{info['id']}]")
                           except websockets.exceptions.ConnectionClosed:
                               clients.pop(client, None)
               
               except json.JSONDecodeError:
                   print(f"[{client_id}] Invalid JSON: {message}")
       
       except websockets.exceptions.ConnectionClosed:
           print(f"[{client_id}] Disconnected")
       finally:
           del clients[websocket]
           print(f"  Remaining: {len(clients)}")
   ```

4. Main function:
   ```python
   async def main():
       print("=" * 50)
       print("Knockout Tournament WebSocket Server")
       print("=" * 50)
       print("Port: 8081")
       print("Commands: init, score, settle, reset")
       print("Press Ctrl+C to stop\n")
       
       async with websockets.serve(handler, "localhost", 8081):
           await asyncio.Future()
   ```

5. Entry point:
   ```python
   if __name__ == "__main__":
       try:
           asyncio.run(main())
       except KeyboardInterrupt:
           print("\nServer stopped")
   ```

2. WebSocket connection (adapted from BPL):
   ```javascript
   connectWebSocket() {
     this.updateConnectionStatus('connecting');
     
     try {
       this.ws = new WebSocket(this.wsUrl);
       
       this.ws.onopen = () => {
         console.log('WebSocket connected');
         this.updateConnectionStatus('connected');
       };
       
       this.ws.onmessage = (event) => {
         try {
           const message = JSON.parse(event.data);
           this.handleCommand(message);
         } catch (error) {
           console.error('Invalid message:', error);
         }
       };
       
       this.ws.onclose = () => {
         console.log('WebSocket disconnected');
         this.updateConnectionStatus('disconnected');
         setTimeout(() => this.connectWebSocket(), this.reconnectInterval);
       };
       
       this.ws.onerror = (error) => {
         console.error('WebSocket error:', error);
         this.updateConnectionStatus('error');
       };
     } catch (error) {
       console.error('Connection failed:', error);
       setTimeout(() => this.connectWebSocket(), this.reconnectInterval);
     }
   }
   
   updateConnectionStatus(status) {
     this.els.statusDot.className = 'status-dot ' + status;
     const texts = {
       connected: '已连接',
       connecting: '连接中...',
       disconnected: '离线',
       error: '连接错误'
     };
     this.els.statusText.textContent = texts[status] || '未知';
   }
   ```

3. Command dispatcher:
   ```javascript
   handleCommand(message) {
     if (!message.cmd) {
       console.error('Missing cmd field:', message);
       return;
     }
     
     console.log('Received:', message.cmd);
     
     switch (message.cmd) {
       case 'init':
         this.handleInit(message.data);
         break;
       case 'score':
         this.handleScore(message.data);
         break;
       case 'settle':
         this.handleSettle(message.data);
         break;
       case 'reset':
         this.handleReset();
         break;
       default:
         console.warn('Unknown command:', message.cmd);
     }
   }
   ```
