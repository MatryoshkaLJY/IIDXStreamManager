```javascript
handleInit(data) {
  if (!data || !data.groups) {
    console.error('INIT: Missing groups data');
    return;
  }
  
  // Store tournament name
  this.tournamentState.name = data.tournamentName || '16人淘汰赛';
  if (this.els.tournamentName) {
    this.els.tournamentName.textContent = this.tournamentState.name;
  }
  
  // Reset state
  this.tournamentState.currentStage = 'quarterfinal';
  
  // Initialize quarterfinal groups
  ['A', 'B', 'C', 'D'].forEach(groupName => {
    const players = data.groups[groupName] || [];
    this.tournamentState.groups[groupName] = {
      players: players.map((name, index) => ({
        name: name,
        position: index + 1,
        rawScores: [], // Array of 4 song scores
        points: 0,     // Total points
        totalRawScore: 0
      })),
      currentRound: 0,
      settled: false,
      advancing: []
    };
    
    // Update DOM with player names
    players.forEach((name, index) => {
      this.updatePlayerNode(groupName, index + 1, { name });
      this.setPlayerState(groupName, index + 1, 'active');
    });
  });
  
  // Initialize empty semifinal and final groups
  ['AB', 'CD'].forEach(groupName => {
    this.tournamentState.groups[groupName] = {
      players: [],
      currentRound: 0,
      settled: false,
      advancing: []
    };
  });
  
  this.tournamentState.finals = {
    players: [],
    currentRound: 0,
    settled: false,
    champion: null
  };
  
  // Clear any existing path lighting
  document.querySelectorAll('.connection-path').forEach(path => {
    path.classList.remove('lit', 'champion-path');
  });
  
  // Reset champion display
  if (this.els.championName) {
    this.els.championName.textContent = '';
  }
  
  console.log('Tournament initialized:', this.tournamentState.name);
}
```

```javascript
handleScore(data) {
  if (!data || !data.stage || !data.group || !data.scores) {
    console.error('SCORE: Missing required fields');
    return;
  }
  
  const { stage, group, round, scores } = data;
  
  // Validate round number (1-4)
  if (round < 1 || round > 4) {
    console.error('SCORE: Invalid round number:', round);
    return;
  }
  
  // Get group state
  let groupState;
  if (stage === 'quarterfinal') {
    groupState = this.tournamentState.groups[group];
  } else if (stage === 'semifinal') {
    groupState = this.tournamentState.groups[group]; // AB or CD
  } else if (stage === 'final') {
    groupState = this.tournamentState.finals;
  }
  
  if (!groupState) {
    console.error('SCORE: Invalid stage/group:', stage, group);
    return;
  }
  
  // Update current round
  groupState.currentRound = round;
  
  // Calculate ranks and points for this song
  const sortedScores = [...scores].sort((a, b) => b.score - a.score);
  const scoreMap = new Map(scores.map(s => [s.player, s.score]));
  
  // Update each player's data
  groupState.players.forEach(player => {
    const playerScore = scoreMap.get(player.name);
    if (playerScore !== undefined) {
      // Store raw score for this round
      player.rawScores[round - 1] = playerScore;
      player.totalRawScore = player.rawScores.reduce((a, b) => a + b, 0);
      
      // Calculate rank and points for this song
      const rank = sortedScores.findIndex(s => s.player === player.name) + 1;
      const points = this.calculatePoints(rank);
      
      // Add to total (only count each song once)
      if (!player.roundPoints) player.roundPoints = {};
      player.roundPoints[round] = points;
      player.points = Object.values(player.roundPoints).reduce((a, b) => a + b, 0);
      
      // Update DOM
      const position = player.position || (groupState.players.indexOf(player) + 1);
      this.updatePlayerNode(group, position, {
        score: playerScore,
        points: player.points,
        rank: rank
      });
    }
  });
  
  console.log(`Score recorded for ${stage} ${group}, round ${round}`);
}

calculatePoints(rank) {
  // Points: 1st=2, 2nd=1, 3rd/4th=0
  const pointsTable = { 1: 2, 2: 1, 3: 0, 4: 0 };
  return pointsTable[rank] || 0;
}
```

```javascript
handleSettle(data) {
  if (!data || !data.stage || !data.group) {
    console.error('SETTLE: Missing required fields');
    return;
  }
  
  const { stage, group } = data;
  
  // Get group state
  let groupState;
  let nextStage, nextGroup;
  
  if (stage === 'quarterfinal') {
    groupState = this.tournamentState.groups[group];
    nextStage = 'semifinal';
    nextGroup = (group === 'A' || group === 'B') ? 'AB' : 'CD';
  } else if (stage === 'semifinal') {
    groupState = this.tournamentState.groups[group];
    nextStage = 'final';
    nextGroup = 'final';
  } else if (stage === 'final') {
    groupState = this.tournamentState.finals;
    nextStage = 'champion';
  }
  
  if (!groupState) {
    console.error('SETTLE: Invalid stage/group:', stage, group);
    return;
  }
  
  // Sort players by points (desc), then by total raw score (desc) for tiebreaker
  const sortedPlayers = [...groupState.players].sort((a, b) => {
    if (b.points !== a.points) {
      return b.points - a.points; // Higher points first
    }
    return b.totalRawScore - a.totalRawScore; // Tiebreaker: higher raw score
  });
  
  // Top 2 advance
  const advancing = sortedPlayers.slice(0, 2);
  const eliminated = sortedPlayers.slice(2);
  
  groupState.advancing = advancing.map(p => p.name);
  groupState.settled = true;
  
  // Update UI for eliminated players
  eliminated.forEach(player => {
    const position = player.position || (groupState.players.indexOf(player) + 1);
    this.setPlayerState(group, position, 'eliminated');
  });
  
  // Update UI for advancing players and light paths
  advancing.forEach((player, index) => {
    const position = player.position || (groupState.players.indexOf(player) + 1);
    this.setPlayerState(group, position, 'advancing');
    
    // Light the path to next stage
    const fromKey = `${group}${position}`;
    let toKey;
    
    if (stage === 'quarterfinal') {
      // Map to semifinal position
      const groupOffset = (group === 'A' || group === 'C') ? 0 : 2;
      const posInGroup = (position <= 2) ? 1 : 2;
      toKey = `${nextGroup}${groupOffset + posInGroup}`;
    } else if (stage === 'semifinal') {
      toKey = `F${index + 1}`;
    }
    
    if (toKey) {
      this.lightPath(fromKey, toKey);
    }
  });
  
  // Advance players to next stage
  if (nextStage === 'semifinal') {
    // Add to semifinal group
    const nextGroupState = this.tournamentState.groups[nextGroup];
    advancing.forEach((player, index) => {
      const newPosition = nextGroupState.players.length + 1;
      nextGroupState.players.push({
        name: player.name,
        position: newPosition,
        rawScores: [],
        points: 0,
        totalRawScore: 0,
        prevGroup: group,
        prevPosition: player.position
      });
      
      // Update DOM for semifinal node
      this.updatePlayerNode(nextGroup, newPosition, { name: player.name });
      this.setPlayerState(nextGroup, newPosition, 'active');
    });
  } else if (nextStage === 'final') {
    // Add to finals
    const finalsState = this.tournamentState.finals;
    advancing.forEach((player, index) => {
      const newPosition = finalsState.players.length + 1;
      finalsState.players.push({
        name: player.name,
        position: newPosition,
        rawScores: [],
        points: 0,
        totalRawScore: 0,
        prevGroup: group,
        prevPosition: player.position
      });
      
      // Update DOM for final node
      this.updatePlayerNode('F', newPosition, { name: player.name });
      this.setPlayerState('F', newPosition, 'active');
    });
  } else if (nextStage === 'champion') {
    // Champion determined
    const champion = advancing[0];
    this.tournamentState.finals.champion = champion.name;
    
    // Update champion display
    if (this.els.championName) {
      this.els.championName.textContent = champion.name;
    }
    
    // Light champion path from finals
    const championPosition = champion.position || 1;
    this.lightPath(`F${championPosition}`, 'champion', true);
    
    // Light full progression path
    this.lightChampionPath(champion);
  }
  
  console.log(`${stage} ${group} settled. Advancing:`, groupState.advancing);
}

lightChampionPath(champion) {
  // Trace back and light all paths from quarterfinals to champion
  // This requires tracking the player's progression through stages
  
  // Light quarterfinal to semifinal path
  if (champion.prevGroup && champion.prevPosition) {
    this.lightPath(`${champion.prevGroup}${champion.prevPosition}`, `F${champion.position}`, true);
    
    // Find the quarterfinal origin
    const semiGroup = champion.prevGroup;
    const semiPlayer = this.tournamentState.groups[semiGroup].players.find(
      p => p.name === champion.name
    );
    
    if (semiPlayer && semiPlayer.prevGroup) {
      this.lightPath(`${semiPlayer.prevGroup}${semiPlayer.prevPosition}`, `${semiGroup}${champion.prevPosition}`, true);
    }
  }
}
```

```javascript
handleReset() {
  // Reset state
  this.tournamentState = {
    name: '',
    groups: {
      A: { players: [], scores: {}, settled: false, advancing: [] },
      B: { players: [], scores: {}, settled: false, advancing: [] },
      C: { players: [], scores: {}, settled: false, advancing: [] },
      D: { players: [], scores: {}, settled: false, advancing: [] },
      AB: { players: [], scores: {}, settled: false, advancing: [] },
      CD: { players: [], scores: {}, settled: false, advancing: [] }
    },
    finals: { players: [], scores: {}, settled: false, champion: null },
    currentStage: 'quarterfinal'
  };
  
  // Clear all player nodes
  document.querySelectorAll('.player-node').forEach(node => {
    const nameEl = node.querySelector('.player-name');
    const scoreEl = node.querySelector('.player-score');
    const pointsEl = node.querySelector('.player-points');
    const rankEl = node.querySelector('.player-rank');
    
    if (nameEl) nameEl.textContent = 'TBD';
    if (scoreEl) scoreEl.textContent = '0';
    if (pointsEl) pointsEl.textContent = '0 pts';
    if (rankEl) rankEl.textContent = '-';

  // Clear all path lighting
  document.querySelectorAll('.connection-path').forEach(path => {
    path.classList.remove('lit', 'champion-path');

  // Clear tournament name
  if (this.els.tournamentName) {
    this.els.tournamentName.textContent = '16人淘汰赛';
  }
  
  console.log('Tournament reset');
}
```
